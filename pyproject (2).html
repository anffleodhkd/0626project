colors = ['g','r','b']
linestyles = ['-','-.','--']

def plot_model(x, y, models=None):
    plt.scatter(x, y)
    plt.axis([0,1.6,0,120])
    if models:
        mx = sp.linspace(0,150,num=100)
        for model, color, line in zip(models, colors, linestyles):
            plt.plot(mx, model(mx),linestyle=line, c=color)
        plt.legend(["d=%i" % m.order for m in models], loc='upper left')
    plt.show()
